ESSs#

\begin{equation} Z = R + jX \end{equation}

\begin{equation} Z_n = R_n + jX_n \end{equation}

\begin{equation} Z_{ng} = R_{ng} + jX_{ng} \end{equation}

\[\begin{split} \begin{align} \vec v_{ta} &= v_{tar} +j v_{tai} \\ \vec v_{tb} &= v_{tbr} +j v_{tbi} \\ \vec v_{tc} &= v_{tcr} +j v_{tci} \\ \vec v_{tn} &= v_{tnr} +j v_{tni} \end{align} \end{split}\]

...
vscs:[
     {"type":"ac_3ph_4w","bus":"A1","S_n":100e3,"U_n":400,
      "R":0.01,"X":0.05,"R_n":0.01,"X_n":0.1,"R_ng":0.01,"X_ng":3.0}
     ],
